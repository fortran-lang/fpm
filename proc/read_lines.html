function read_lines(fh) result(lines)
    integer, intent(in) :: fh
    type(string_t), allocatable :: lines(:)

    integer :: i
    integer :: iostat

    allocate(lines(number_of_rows(fh)))
    do i = 1, size(lines)
        call getline(fh, lines(i)%s, iostat)
    end do

end function read_lines
